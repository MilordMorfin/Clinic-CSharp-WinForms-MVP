SELECT CONCAT(pacjenci.imie, " ", pacjenci.nazwisko) AS Pacjent, CONCAT(doktorzy.imie, " ", doktorzy.nazwisko) AS Lekarz, wizyty.data AS "Data wizyty", wizyty.opis AS Rozeznanie, leki.nazwa AS Lek, dawki.ile AS Dawka, dawki_i_leki.od_kiedy AS Od, dawki_i_leki.do_kiedy AS Do FROM `wizyty` JOIN pacjenci ON pacjenci.idp=wizyty.idp JOIN doktorzy ON doktorzy.idd=wizyty.idd JOIN dawki_i_leki ON dawki_i_leki.iddl=wizyty.iddl JOIN dawki ON dawki.idd=dawki_i_leki.idd JOIN leki ON dawki_i_leki.idl=leki.idl;
